let str = 'Hello, World!';

// Using substring
console.log(str.substring(1, 5)); // "ello"
console.log(str.substring(5, 1)); // "ello" (arguments are swapped)
console.log(str.substring(7)); // "World!"

// Using slice
console.log(str.slice(1, 5)); // "ello"
console.log(str.slice(5, 1)); // "" (no swapping of arguments)
console.log(str.slice(-6)); // "World!"


Key Differences b/w substring and slice
1) Swapping of Arguments:

substring swaps startIndex and endIndex if startIndex is greater.
slice does not swap; it returns an empty string if startIndex is greater.

2) Negative Indices:

substring does not support negative indices.
slice supports negative indices, counting from the end of the string.
