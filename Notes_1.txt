// Define a string with spaces and tabs
const originalString =
  'This is a   sample   string     with       spaces and tabs.';

// Apply the replace method to remove spaces and tabs
const stringWithoutSpaces = originalString.replace(/\s+/g, '');

// Output the original and modified strings
console.log('Original string:', originalString);
console.log('String without spaces:', stringWithoutSpaces); //Thisisasamplestringwithspacesandtabs.
